71/*
72 Entity
73
74 An entity is a collection of components that make up a game object.
75*/
76struct ye_entity {
77 bool active; // controls whether system will act upon this entity and its components
78
79 int id; // unique id for this entity
80 char *name; // name that can also be used to access the entity
81
82 struct ye_component_transform *transform; // transform component
83 struct ye_component_renderer *renderer; // renderer component
84 struct ye_component_script *script; // script component
85 struct ye_component_interactible *interactible; // interactible component
86 struct ye_component_camera *camera; // camera component
87 struct ye_component_physics *physics; // physics component
88 struct ye_component_collider *collider; // collider component
89 struct ye_component_tag *tag; // tag component
90
91 /* NOTE/TODO:
92 We can have arrays in the future malloced to the
93 size of struct ye_component_* and then we can have
94 multiple components of the same type on an entity.
95
96 This would be good for tags, but for all else we would actually need
97 to explicitely link which transform and which components are
98 used for other components which have been assuming them by default
99 */
100};
101
102// 2d vector
103struct ye_vec2f {
104 float x;
105 float y;
106};
107
108/*
109 Script component
110
111 Holds information on a script that is attatched to an entity.
112 This script will have its main loop run once per frame.
113*/
114struct ye_component_script {
115 bool active; // controls whether system will act upon this component
116
117 char *script_path;
118};
119
120/*
121 Interactible component
122
123 Holds information on how an entity can be interacted with.
124*/
125struct ye_component_interactible {
126 bool active; // controls whether system will act upon this component
127
128 void *data; // data to communicate when callback finishes
129 void (*callback)(void *data); // callback to run when entity is interacted with
130
131 /*
132 Currently, we will get the transform of an object with a
133 interactible component to check bounds for clicks.
134
135 In the future, we would want a collider_2d component with
136 an assosciated type and relative position so we can more
137 presicely check for interactions. I dont forsee the need
138 for this for a while though. Maybe future me in a year reading
139 this wishes I did it now xD
140 */
141};
