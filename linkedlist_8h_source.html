76/*
77 Throws a new item onto the head
78
79 We dont really have a use case for implementing sorted insertion
80 at the moment, anything using this generic will have negligible
81 item counts
82*/
83void LL_insert(LinkedList *list, void *data, char *key) {
84 Node *newNode = (Node *)malloc(sizeof(Node));
85 newNode->data = data;
86 newNode->next = list->head;
87 newNode->key = key;
88 list->head = newNode;
89}

91/*
92 Traverses the list, runs a passed callback fn ptr for every item,
93 passing each item as a parameter
94*/
95void LL_traverse(LinkedList *list, void (*process)(void *)) {
96 Node *current = list->head;
97 while (current != NULL) {
98 process(current);
99 current = current->next;
100 }
101}

103/*
104 Attempts to delete an item from the list by key
105*/
106void LL_delete(LinkedList *list, char *key){
107 Node *current = list->head;
108 Node *prev = NULL;
109 while (current != NULL) {
110 if(current->key == key){
111 if(prev == NULL){
112 list->head = current->next;
113 } else {
114 prev->next = current->next;
115 }
116 free(current);
117 return;
118 }
119 prev = current;
120 current = current->next;
121 }
122}

124/*
125 Frees a linked list
126*/
127void LL_freeList(LinkedList *list) {
128 Node *current = list->head;
129 while (current != NULL) {
130 Node *temp = current;
131 current = current->next;
132 free(temp);
133 }
134 list->head = NULL;
135}
