24/*
25 Enum to store different unique types of renderers.
26 This is how we identify steps needed to render different types of entities.
27
28 Ex: animation renderer knows it needs to increment frames, text renderer knows how to reconstruct text, etc
29*/
30enum ye_component_renderer_type {
31 YE_RENDERER_TYPE_TEXT,
32 YE_RENDERER_TYPE_TEXT_OUTLINED,
33 YE_RENDERER_TYPE_IMAGE,
34 YE_RENDERER_TYPE_ANIMATION
35};
